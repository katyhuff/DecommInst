15 /// This Institution is specifically designed for market-driven
16 /// building/decommissioning of facilities. It utlizes manager mixin classes to
17 /// assist in this task, and was created for the specific purpose of runing the
18 /// fco code-to-code comparison. However, it was designed to be sufficiently
19 /// generic for general use.

22 /// loaded by the Agent class when requested.

24 /// @section intro Introduction

26 /// material criteria occurs in the markets.

78  virtual void InfileToDb(cyclus::InfileTree* tree, cyclus::DbInit di) {
79  cyclus::Institution::InfileToDb(tree, di);
80  tree = tree->SubTree("config/*");
81  cyclus::InfileTree* sub;
82  int i;
83  int n;
84  target_fac = cyclus::Query<std::string>(tree, "target_fac");
85  num_to_build = cyclus::Query<int>(tree, "num_to_build");
86  target_commod = cyclus::Query<std::string>(tree, "target_commod");
87  sub = tree->SubTree("prototypes");
88  n = sub->NMatches("val");
89  prototypes.resize(n);
90  for (i = 0; i < n; ++i) {
91  prototypes[i] = cyclus::Query<std::string>(sub, "val", i);
92  }
93  amt_req = cyclus::Query<double>(tree, "amt_req");
94  di.NewDatum("Info")
95  ->AddVal("target_fac", target_fac)
96  ->AddVal("num_to_build", num_to_build)
97  ->AddVal("target_commod", target_commod)
98  ->AddVal("prototypes", prototypes)
99  ->AddVal("amt_req", amt_req)
100  ->Record();
